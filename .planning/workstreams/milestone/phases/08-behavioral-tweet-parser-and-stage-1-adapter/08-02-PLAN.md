```python
def _make_twibot_json(tmp_path, n: int = 5) -> str:
    """Write synthetic TwiBot-20 JSON to tmp_path and return its path.

    Each record gets a 3-tweet `tweet` list (1 RT, 1 MT, 1 original) so the
    behavioral adapter has data to classify. The `profile` dict carries only
    fields load_accounts() requires; the column adapter no longer reads
    profile.followers_count / friends_count / listed_count (FEAT-02).
    """
    records = []
    for i in range(n):
        records.append({
            "ID": f"tw_{i:03d}",
            "label": "1" if i % 2 else "0",
            "profile": {
                "screen_name": f"user_{i}",
                "statuses_count": "0",
                "followers_count": "0",
                "friends_count": "0",
                "created_at": "Mon Jan 01 00:00:00 +0000 2020",
            },
            "tweet": [
                f"RT @target_{i}: amplified content from user {i}",
                f"MT @other_{i}: modified retweet from user {i}",
                f"original thought number {i}",
            ],
            "neighbor": None,
        })
    path = str(tmp_path / "test.json")
    with open(path, "w", encoding="utf-8") as f:
        json.dump(records, f)
    return path
```
