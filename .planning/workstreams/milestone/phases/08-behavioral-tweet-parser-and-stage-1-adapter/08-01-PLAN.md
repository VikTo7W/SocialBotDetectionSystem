```python
def parse_tweet_types(messages: List[Dict[str, Any]]) -> Dict[str, Any]:
    """Classify an account's tweets into RT, MT, and original buckets.

    Implements the FEAT-01 behavioral tweet parser per phase 8 decisions
    D-01 (case-insensitive prefix check, no regex), D-02 (best-effort
    space-split @-token extraction), D-03 (case-insensitive matching),
    and D-05 (return dict shape).

    Parameters
    ----------
    messages : List[Dict[str, Any]]
        Per-account messages list as produced by load_accounts(); each entry
        is ``{"text": str, "ts": None, "kind": "tweet"}``.

    Returns
    -------
    Dict[str, Any]
        Dict with keys:
          - rt_count (int): tweets whose stripped+uppercased text starts with "RT "
          - mt_count (int): tweets whose stripped+uppercased text starts with "MT "
          - original_count (int): all other tweets
          - rt_mt_usernames (list[str]): distinct lowercased @-handles extracted
            from RT/MT tweets, in insertion order (first-seen wins).

    Notes
    -----
    Username extraction strips the leading "@" and any trailing ":" from the
    first space-delimited token after the RT/MT prefix. Tweets with no
    @-token after the prefix still count toward rt_count/mt_count but
    contribute no username (D-02). Empty messages list returns all-zero dict
    with empty usernames list (D-11).
    """
    rt_count = 0
    mt_count = 0
    original_count = 0
    rt_mt_usernames: List[str] = []

    for msg in messages:
        text = msg["text"].strip()
        upper = text.upper()
        if upper.startswith("RT "):
            rt_count += 1
            tokens = text.split()
            if len(tokens) > 1 and tokens[1].startswith("@"):
                rt_mt_usernames.append(tokens[1].lstrip("@").rstrip(":").lower())
        elif upper.startswith("MT "):
            mt_count += 1
            tokens = text.split()
            if len(tokens) > 1 and tokens[1].startswith("@"):
                rt_mt_usernames.append(tokens[1].lstrip("@").rstrip(":").lower())
        else:
            original_count += 1

    return {
        "rt_count": rt_count,
        "mt_count": mt_count,
        "original_count": original_count,
        "rt_mt_usernames": list(dict.fromkeys(rt_mt_usernames)),
    }
```
